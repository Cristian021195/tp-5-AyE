ESPECIFICACION ALGEBRAICA DE TIPO PILA

SINTAXIS
    PILAVACIA -> PILA #
    ESPILAVACIA: PILA -> BOOL
    TOP: PILA -> ITEM U {indefinido}
    POP: PILA -> PILA
    PUSH: PILA x ITEM -> PILA #
    PUSHF: PILA x ITEM -> PILA
    APILA: PILA x PILA -> PILA
    APILAINV: PILA x PILA -> PILA
    FONDO: PILA -> ITEM U {indefinido}
    POPFONDO: PILA -> PILA
    ESSIMETRICA: PILA -> BOOL    

SEMENTAICA: ∀ P1, P2 ∈ PILA, ∀ x,b,y ∈ ITEM 
    ESPILAVACIA(PILAVACIA) = true
    ESPILAVACIA(PUSH(P1, x)) = false
    TOP(PILAVACIA) = indefinido
    TOP(PUSH(P1, x)) = x
    POP(PILAVACIA) = PILAVACIA
    POP(PUSH(P1, x)) = P1
    PUSHF(PILAVACIA, x) = PUSH(PILAVACIA,x)
    PUSHF(PUSH(P1,x), b) = PUSH(PUSHF(P1,b),x)

    FONDO(PILAVACIA) = indefinido
    FONDO(PUSH(P1, x)) = FONDO(P1)

    POPFONDO(PILAVACIA) = PILAVACIA
    POPFONDO(PUSH(P1, x)) = PUSH(POPFONDO(P1),x)

    ESSIMETRICA(PILAVACIA) = true
    ESSIMETRICA(PUSH(P1, x)) = true
    ESSIMETRICA(PUSH(PUSH(P1, x),b)) = 
        SI x = b ENTONCES true
        SINO false
    ESSIMETRICA(PUSH(PUSH(PUSH(P1, x),b),y)) = 
        SI x = y ENTONCES
            ESSIMETRICA(P1) ?
        SINO FALSE;


    APILA(PILAVACIA, PILAVACIA) = PILAVACIA
    APILA(PUSH(P1,x), PILAVACIA) = PUSH(P1,x)
    APILA(PILAVACIA, PUSH(P2,b)) = PUSH(P2,b)
    APILA(PUSH(P1,x), PUSH(P2,b)) = ...
        PUSH(APILA(PUSH(P1,x), P2),b)

    APILAINV(PILAVACIA, PILAVACIA) = PILAVACIA
    APILAINV(PUSH(P1,x), PILAVACIA) = PUSH(P1, x)
    APILAINV(PILAVACIA, PUSH(P2,b)) = APILAINV(PUSH(PILAVACIA, b), P2)
    APILAINV(PUSH(P1,x), PUSH(P2,b)) = ... 
        APILAINV(PUSH(PUSH(P1,x), b), P2)
